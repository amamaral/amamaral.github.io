def smooth_segment(ti, tf, xi, vi, xf, vf, interp_points=50):
    t = linspace(ti, tf, interp_points)

    dx = xf-xi
    dt = tf-ti

    c1 = (3*dx/dt - vf - 2*vi)/dt
    c2 = (2*dx/dt - vf - vi)/dt**2

    x = xi + vi*(t-ti) + c1 * (t-ti)**2 - c2 * (t-ti)**3
    
    return t, x

def interpolate_curve(t_arr, x_arr, interp_points = 10, vfinal=None):
    # Forward difference velocity
    v = (x_arr[1:]-x_arr[:-1])/(t_arr[1:]-t_arr[:-1])
    if vfinal is None:
        v = append(v, v[-1])
    else:
        v = append(v, vfinal)
    
    t_new = zeros(len(t_arr)*interp_points)
    x_new = zeros(len(t_arr)*interp_points)
    
    for i, (ti, xi) in enumerate(zip(t_arr, x_arr)):
        if i<len(t_arr)-1:
            # Smooth interpolation between xi and xip1, assuming vi and vf as the initial and final speeds, respect.
            # The eq. below assumes an acceleration given by a=a(t)=a0+a1*t

            tip1 = t_arr[i+1]
            xip1 = x_arr[i+1]
            
            dx = xip1 - xi
            dt = tip1 - ti
            
            vi = v[i]
            vf = v[i+1]
            
            c1 = (3*dx/dt - vf - 2*vi)/dt
            c2 = (2*dx/dt - vf - vi)/dt**2
            
            t_interp = linspace(ti, tip1, interp_points)
            x_interp = xi + vi*(t_interp-ti) + c1 * (t_interp-ti)**2 - c2 * (t_interp-ti)**3
            
            for j, (tj, xj) in enumerate(zip(t_interp, x_interp)):
                t_new[interp_points * i + j] = tj
                x_new[interp_points * i + j] = xj

    return t_new[:-interp_points], x_new[:-interp_points]

from matplotlib.patches import FancyArrowPatch
from mpl_toolkits.mplot3d import proj3d

class Arrow3D(FancyArrowPatch):

    def __init__(self, xs, ys, zs, *args, **kwargs):
        FancyArrowPatch.__init__(self, (0, 0), (0, 0), *args, **kwargs)
        self._verts3d = xs, ys, zs

    def draw(self, renderer):
        xs3d, ys3d, zs3d = self._verts3d
        xs, ys, zs = proj3d.proj_transform(xs3d, ys3d, zs3d, renderer.M)
        self.set_positions((xs[0], ys[0]), (xs[1], ys[1]))
        FancyArrowPatch.draw(self, renderer)

a = Arrow3D([-4, 1], [-4, 1], [-2, 1], mutation_scale=20,
            lw=1, arrowstyle="-|>", color="k")
ax.add_artist(a)

plt.show()
